import Vue from 'vue'
import App from './App'/*引入App这个组件*/
import router from './router'/*引入路由配置*/

Vue.config.productionTip = false

//使用 eslint-disable + 规则名 的作用是不检测这条规则，注意要使用 eslint-enable 结束哟
//直接 eslint-disable 的作用是完全禁用ESLint进行检测

/* eslint-disable no-new */
new Vue({
  el: '#app',/*最后效果将会替换页面中id为app的div元素*/
  router,/*使用路由*/
  template: '<App/>',/*告知页面这个组件用这样的标签来包裹着,并且使用它*/
  components: { App }/*告知当前页面想使用App这个组件*/
})

路由配置

import vue from 'vue'
import vueRouter from 'vue-router'

vue.use(vueRouter)

import hello from '@/components/hello'
//引入需要使用模块

var router = new vueRouter({
  mode: 'history', //默认为hash模式，history，hash
  linkActiveClass: 'isCheck',  
  //链接点击被触发后的class;默认为：router-link-active
  routes: [
    {
      path: '*',
      //###页面不存在时跳转到404页面###
      component: page404,
      
      // ###重定向###
      /*访问页面不存在时定向到hello 
      redirect:'/hello'          //写法1
      redirect:{path:'/hello'}   //写法2
      redirect:{name:'hello'}   //写法3 ==>  name属性
      redirect(linkMess){
        // linkMess目标路由对象，访问的路径信息
        console.log(linkMess)
        return '/hello'
      }
      */
    },
    {
      path: '/hello',
      name: 'hello',
      component: hello,
      alias:'/index'  //设置别名，访问index路径同样显示hello页面
    },
    {
      path: '/about',
      component: about,
      children: [
        {
          path: '',  //空为默认打开的子路由
          name: 'about', //如果有默认子路由的情况下，那么父路由不需要name属性
          component: me
        }
      ]
    }
    
  ]
})

export default router
//导出

{
path: 路径
name: 路由名
component: 对应组件
children:子路由
}
